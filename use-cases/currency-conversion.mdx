## Initial Steps

- Setup an account (if you don't have an account already)
- Get your API key from [here](https://dash.jigsawstack.com/).

```go convert.go
package main
import (
    "fmt"
    "net/http"
    "io/ioutil"
)
func main() {


    headers := make(map[string]string)
    headers["Content-Type"] = "application/json"
    headers["x-api-key"] = "<your-api-key-here>"

    url := fmt.Sprintf("http://api.jigsawstack.com/v1/currency/convert?from=USD&to=NGN&amount=100")
    req, err := http.NewRequest("GET", url, nil)
    if err != nil {
        fmt.Println(err)
        return
    }

    for key, value := range headers {
        req.Header.Add(key, value)
    }

    resp, err := http.DefaultClient.Do(req)
    if err != nil {
        fmt.Println(err)
        return
    }

    defer resp.Body.Close()

    body, err := ioutil.ReadAll(resp.Body)
    if err != nil {
        fmt.Println(err)
        return
    }

    fmt.Println(string(body))
}

```
