### Body

<ParamField body="subject" type="string" required>

</ParamField>

<ParamField body="webhook" type="string">

</ParamField>

<ParamField body="to" type="Array<string>">

</ParamField>

<ParamField body="identifier" type="string">
  The identifier of the email hook.
</ParamField>

<ParamField body="meta" type="object">
  The meta data of the email hook.
</ParamField>
