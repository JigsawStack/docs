---
title: "HTML To Any"
api: "POST https://api.jigsawstack.com/v1/web/html_to_any"
description: "Capture screenshots of a website, convert HTML to both image and PDF formats."
---

### Body

<ParamField body="html" type="string">
  The HTML to convert to image.
</ParamField>

<ParamField body="url" type="string">
  The HTML to convert to image.
</ParamField>

<ParamField body="goto_options" type="object">
  This parameter allows you to specify custom page-load behavior.
  <Expandable title="property">
    <ResponseField name="timeout" type="number" default="15000">
      Custom timeout in milliseconds.
    </ResponseField>
    <ResponseField name="wait_until" type="string" default="load">
      <ul>
        <li>load</li>
        <li>domcontentloaded</li>
        <li>networkidle0</li>
        <li>networkidle2</li>
      </ul>
    </ResponseField>
  </Expandable>
</ParamField>
<ParamField body="scale" type="number" initialValue="1" default="1">
  The device scale factor. The default value is 1.
</ParamField>
<ParamField body="full_page" type="boolean">
  Whether to capture the full page.
</ParamField>

<ParamField body="omit_background" type="boolean">
  Whether to omit the background.
</ParamField>

<ParamField body="quality" type="number" default="75">
  The quality of the image. 0-100.
</ParamField>

<ParamField body="type" type="string" default="png">
  The type of the image. Can be one of the following:
  <ul>
    <li>jpeg</li>
    <li>png</li>
    <li>webp</li>
    <li>pdf</li>
  </ul>
</ParamField>

<ParamField body="width" type="number" default="1920">
  The width of the image.
</ParamField>

<ParamField body="height" type="number" default="1080">
  The height of the image.
</ParamField>

<ParamField body="size_preset" type="string">
  The screen size preset to use. Supported values are:
  <ul>
    <li>`QVGA`</li>
    <li>`VGA`</li>
    <li>`SVGA`</li>
    <li>`HD`</li>
    <li>`HD+`</li>
    <li>`SXGA`</li>
    <li>`FHD`</li>
    <li>`2K`</li>
    <li>`2K QHD`</li>
    <li>`5K`</li>
    <li>`4K UHD`</li>
  </ul>
  See the full list of screen size presets
  [here](/docs/additional-resources/size-preset)
</ParamField>

<ParamField body="pdf_display_header_footer" type="boolean"></ParamField>

<ParamField body="pdf_print_background" type="boolean"></ParamField>
<ParamField body="pdf_page_range" type="string"></ParamField>

<ParamField body="is_mobile" type="boolean">
  Sets whether the meta viewport tag is taken into account.
</ParamField>
<ParamField body="dark_mode" type="boolean">
  Sets the color scheme to dark mode.
</ParamField>
<ParamField body="use_graphic_renderer" type="boolean" default="false">
  Enables webgl, gpu and other 3d-apis. This option when enabled, can impact
  performance and increase api latency.
</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<RequestExample>

```javascript JavaScript
import { JigsawStack } from "jigsawstack";

const jigsawstack = JigsawStack({
  apiKey: "your-api-key",
});

const result = await jigsawstack.web.html_to_any({
  html: "<div>hello</div>",
  quality: 100,
  full_page: true,
  omit_background: false,
  type: "jpeg",
});

const data = await result.blob();
```

```python Python
from jigsawstack import JigsawStack

jigsawstack = JigsawStack(api_key="your-api-key")

result = jigsawstack.web.html_to_any({
  "html": "<div>hello</div>",
  "quality": 100,
  "full_page": true,
  "omit_background": false,
  "type": "jpeg",
})
```

```bash cURL
curl --location --request POST 'https://api.jigsawstack.com/v1/web/html_to_any' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <your-api-key>' \
--data '{
    "html": "<div>hello</div>",
      "quality": 100,
      "full_page": true,
      "omit_background": false,
      "type": "jpeg"
}'
```
