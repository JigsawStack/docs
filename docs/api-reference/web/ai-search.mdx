### Body

<ParamField body="query" type="string" required>
  The search value. The maximum query character length is 400.
</ParamField>

<ParamField body="ai_overview" type="boolean" default="true">
  Include AI powered overview in the search results
</ParamField>

<ParamField body="safe_search" type="string" default="moderate">
  Include offensive results in the search results
  <ul>
    <li>`moderate`</li>
    <li>`strict`</li>
    <li>`off`</li>
  </ul>
</ParamField>

<ParamField body="spell_check" type="boolean" default="true">
  Spell check the search query.
</ParamField>

<ParamField body="byo_urls" type="string[]">
  Return results from the specified URLs.
</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="query" type="string">
  The search query that was processed.
</ResponseField>

<ResponseField name="ai_overview" type="string" optional>
  AI-powered overview of the search results (when ai_overview is enabled).
</ResponseField>

<ResponseField name="spell_fixed" type="boolean">
  Indicates whether the search query was spell-corrected.
</ResponseField>

<ResponseField name="is_safe" type="boolean">
  Indicates whether the search results are considered safe based on safe_search settings.
</ResponseField>

<ResponseField name="results" type="array">
  Array of search result objects.
  
  <Expandable title="Result Object Properties">
    <ResponseField name="title" type="string">
      The title of the search result.
    </ResponseField>
    
    <ResponseField name="url" type="string">
      The URL of the search result.
    </ResponseField>
    
    <ResponseField name="description" type="string">
      Brief description of the search result.
    </ResponseField>
    
    <ResponseField name="content" type="string | object">
      The content of the result, either as plain text or structured object with text, image_urls, and links.
    </ResponseField>
    
    <ResponseField name="is_safe" type="boolean">
      Indicates whether this specific result is considered safe.
    </ResponseField>
    
    <ResponseField name="site_name" type="string">
      The name of the website.
    </ResponseField>
    
    <ResponseField name="site_long_name" type="string">
      The full/long name of the website.
    </ResponseField>
    
    <ResponseField name="age" type="string">
      The age/freshness of the content.
    </ResponseField>
    
    <ResponseField name="language" type="string">
      The language of the content.
    </ResponseField>
    
    <ResponseField name="favicon" type="string">
      URL to the website's favicon.
    </ResponseField>
    
    <ResponseField name="snippets" type="string[]">
      Array of relevant text snippets from the result.
    </ResponseField>
    
    <ResponseField name="related_index" type="RelatedIndex[]">
      Array of related search indices.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="image_urls" type="string[]">
  Array of image URLs found in the search results.
</ResponseField>

<ResponseField name="links" type="string[]">
  Array of links found in the search results.
</ResponseField>

<ResponseField name="geo_results" type="array">
  Array of geographic/location-based search results.
  
  <Expandable title="Geo Result Object Properties">
    <ResponseField name="type" type="string">
      The type of geographic result.
    </ResponseField>
    
    <ResponseField name="full_address" type="string">
      Complete address of the location.
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Name of the location.
    </ResponseField>
    
    <ResponseField name="name_preferred" type="string">
      Preferred name for the location.
    </ResponseField>
    
    <ResponseField name="place_formatted" type="string">
      Formatted place name.
    </ResponseField>
    
    <ResponseField name="postcode" type="string" optional>
      Postal/ZIP code of the location.
    </ResponseField>
    
    <ResponseField name="district" type="string" optional>
      District of the location.
    </ResponseField>
    
    <ResponseField name="place" type="string" optional>
      Place/city name.
    </ResponseField>
    
    <ResponseField name="region" type="object" optional>
      Region information (excluding mapbox_id).
    </ResponseField>
    
    <ResponseField name="country" type="object" optional>
      Country information (excluding mapbox_id).
    </ResponseField>
    
    <ResponseField name="language" type="string">
      Language of the location data.
    </ResponseField>
    
    <ResponseField name="geoloc" type="object">
      Geographic coordinates object with type and coordinates array.
    </ResponseField>
    
    <ResponseField name="poi_category" type="string" optional>
      Point of interest category.
    </ResponseField>
    
    <ResponseField name="additional_properties" type="any" optional>
      Additional properties for the location.
    </ResponseField>
  </Expandable>
</ResponseField>
