```javascript Node.js
import { JigsawStack } from "jigsawstack";

const jigsawstack = JigsawStack({ apiKey: "your-api-key" });

const key = "file.png";

const result = await jigsawstack.store.retrieve(key);
const data = result.blob();
```

```python Python
from jigsawstack import JigsawStack

jigsawstack = JigsawStack(api_key="your-api-key")

try:
    response = jigsawstack.store.get("file.png")
    with open("downloaded_image.jpeg", "wb") as image_file:
        image_file.write(response.content)
    print("Image downloaded and saved successfully.")
except JigsawStackError as err:
    print(f"An error occurred: {err}")

```
