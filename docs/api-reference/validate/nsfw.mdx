### Params

<ParamField body="url" type="string">
  The image url. Not required if `file_store_key` is specified.
</ParamField>

<ParamField body="file_store_key" type="string">
  The key used to store the image on Jigsawstack file
  [Storage](/docs/api-reference/store/file/add). Not required if `url` is
  specified.
</ParamField>

### Response

<BasicResponse />

<ResponseField name="nsfw" type="boolean">
  Indicates whether NSFW content was detected in the image.
</ResponseField>


<ResponseField name="nudity" type="boolean">
  Indicates whether nudity was detected in the image.
</ResponseField>

<ResponseField name="gore" type="boolean">
  Indicates whether gore/violence was detected in the image.
</ResponseField>

<ResponseField name="nsfw_score" type="number">
  Confidence score for NSFW content detection (0-1 scale).
</ResponseField>

<ResponseField name="nudity_score" type="number">
  Confidence score for nudity detection (0-1 scale).
</ResponseField>

<ResponseField name="gore_score" type="number">
  Confidence score for gore/violence detection (0-1 scale).
</ResponseField>
