---
title: "MX Record"
api: "GET https://api.jigsawstack.com/v1/validate/mx_record"
description: "Get MX records of any domain in realtime and high accuracy."
---

### Query

<ParamField query="domain" type="string" required>
  The domain to validate the MX records.
</ParamField>

<Snippet file="header.mdx" />

### Response

<ResponseField name="has_mx_record" type="boolean">
  Indicates whether the domain has a valid MX record.
</ResponseField>

<ResponseField name="records" type="record Object">
  <Expandable title="properties">
    <ResponseField name="host" type="string">
      The host.
    </ResponseField>
    <ResponseField name="prefix" type="number">
      The prefix.
    </ResponseField>
  </Expandable>
</ResponseField>
