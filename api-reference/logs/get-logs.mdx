---
title: "Get List of Logs"
api: "GET https://api.jigsawstack.com/v1/logs/list"
description: "List all logs in your project."
---

### Query

<ParamField query="channel_name" type="string">
  The channel name to retrieve logs from.
</ParamField>

<ParamField query="event" type="string">
  The event to retrieve its logs.
</ParamField>

<ParamField query="limit" default="10" type="number">
  The maximum number of logs to be returned. Default is `10` Max is `100`.
</ParamField>

<ParamField query="page" default="0" type="number">
  The page to return.
</ParamField>

<ParamField query="from" type="string">
  The date to filter the results from. Format `YYYY-MM-DD`
</ParamField>

<ParamField query="to" type="string">
  The date to filter the results. Format `YYYY-MM-DD`
</ParamField>

<ResponseField name="success" type="boolean">

<ResponseField name="list" type="array">

<ResponseField name="page" type="number">
  The page number returned.
</ResponseField>

<ResponseField name="limit" type="number">

</ResponseField>
