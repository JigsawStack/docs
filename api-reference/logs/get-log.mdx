---
title: "Get Log"
api: "GET https://api.jigsawstack.com/v1/logs/{id}"
description: "Get log details."
---

### Path

<ParamField path="id" type="string" required>
  The id of the log you want to retrieve its information.
</ParamField>

<Snippet file="header.mdx" />

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="message" type="string">
  The message returned by the API.
</ResponseField>

<ResponseField name="id" type="string">
  Log id
</ResponseField>

<ResponseField name="event" type="string">
  The event name
</ResponseField>

<ResponseField name="channel_name" type="string">
  The channel name.
</ResponseField>

<ResponseField name="description" type="string">
  The event description.
</ResponseField>

<ResponseField name="description_parser" type="string">
  The description parser.
</ResponseField>

<ResponseField name="platform_key" type="string">
  The platform key.
</ResponseField>

<ResponseField name="meta" type="object">
  Set of key-value pairs that you can attach to the log object. This can be
  useful for storing additional information about the object in a structured
  format.
</ResponseField>

<ResponseField name="geo_event_log" type="JSON">
  The geo information.
</ResponseField>
