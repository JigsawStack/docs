---
title: "Update Channel"
api: "POST https://api.jigsawstack.com/v1/logs/channel/{name}"
description: "Update Log Channel."
---

### Body

<ParamField path="name" type="string" required>
  The name of the channel.
</ParamField>

<ParamField body="new_name" type="string">
  The new name of the channel.
</ParamField>

<ParamField body="description" type="string">
  A description of the channel.
</ParamField>

<ParamField body="tags" type="array<string>">
  A list of tags to associate with the channel.
</ParamField>

<ParamField body="emails" type="array<string>">
  A list of email addresses to notify when a new log is added to the channel.
</ParamField>

<ParamField body="webhook" type="string">
  A webhook URL to notify when a new log is added to the channel.
</ParamField>

<Snippet file="header.mdx" />

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="channel_name" type="string">
  The name of the channel.
</ResponseField>

curl --location --request POST 'https://api.jigsawstack.com/v1/logs/channel/:name' \
--header 'Content-Type: application/json' \

--data '{
    "name": "My Channel",

    }
