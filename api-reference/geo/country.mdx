---
title: "Country"
api: "GET https://api.jigsawstack.com/v1/geo/country"
description: "Get geo information for a country"
---

### Query

<ParamField query="country_code" type="string">
  The country code of the country. Specify to filter by country code.
</ParamField>

<ParamField query="currency_code" type="string">
  The currency code of the country to
</ParamField>

<ParamField query="city_code" type="string">
  The city code of the city.
</ParamField>

<ParamField query="search_value" type="string">
  The search value of the city.
</ParamField>

<ParamField query="lat" type="string">
  The latitude of the country.
</ParamField>

<ParamField query="lng" type="string">
  The longitude of the country.
</ParamField>

<ParamField query="limit" type="string" default="10">
  The limit on the items to return
</ParamField>
