### Query

<ParamField query="search_value" type="string" required>
  The search value.
</ParamField>

<ParamField query="country_code" type="string">
  The ISO 3 country code to filter search results.
</ParamField>

<ParamField query="proximity_lat" type="string">
  The proximity latitude of the search value.
</ParamField>

<ParamField query="proximity_lng" type="string">
  The proximity longitude of the search value.
</ParamField>

<ParamField query="types" type="string">
  <ul>
    <li>`address`</li>
    <li>`street`</li>
    <li>`neighborhood`</li>
    <li>`country`</li>
    <li>`region`</li>
    <li>`place`</li>
    <li>`street`</li>
    <li>`postcode`</li>
    <li>`district`</li>
    <li>`city`</li>
  </ul>
</ParamField>

<Snippet file="header.mdx" />

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<RequestExample>

```javascript Node.js
import { JigsawStack } from "jigsawstack";

const jigsawstack = JigsawStack({
  apiKey: "your-api-key",
});

const result = await jigsawstack.geo.search({
  search_value: "Times square",
});
```

```bash cURL
curl --location --request GET 'https://api.jigsawstack.com/v1/geo/search?search_value=Times square' \
--header 'x-api-key: <your-api-key>'
```

</RequestExample>
