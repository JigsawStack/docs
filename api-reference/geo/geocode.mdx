---
title: "Geocode"
api: "GET https://api.jigsawstack.com/v1/geo/geocode"
description: "Search for addresses or locations by name (forward search) or coordinates (reverse search)."
---

### Query

<ParamField query="search_value" type="string">
  The search value. Not required if `lat` and `lng` are specified.
</ParamField>

<ParamField query="lat" type="string">
  The latitude of the address of interest. Not required if `search_value` is
  specified.
</ParamField>

<ParamField query="lng" type="string">
  The longitude of the address of interest. Not required if `search_value` is
  specified.
</ParamField>

<ParamField query="country_code" type="string">
  The ISO 3 country code to filter search results.
</ParamField>

<ParamField query="proximity_lat" type="string">
  The proximity latitude of the search value.
</ParamField>

<ParamField query="proximity_lng" type="string">
  The proximity longitude of the search value.
</ParamField>

<ParamField query="types" type="string">
  <ul>
    <li>`address`</li>
    <li>`street`</li>
    <li>`neighborhood`</li>
    <li>`country`</li>
    <li>`region`</li>
    <li>`place`</li>
    <li>`street`</li>
    <li>`postcode`</li>
    <li>`district`</li>
    <li>`city`</li>
  </ul>
</ParamField>
