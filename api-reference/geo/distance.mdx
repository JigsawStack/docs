### Query

<ParamField query="lat1" type="string">
  The latitude of the first point.
</ParamField>

<ParamField query="lat2" type="string">
  The latitude of the second point.
</ParamField>

<ParamField query="lng1" type="string">
  The longitude of the first point.
</ParamField>

<ParamField query="lng2" type="string">
  The longitude of the second point.
</ParamField>

<ParamField query="lng" type="string">
  The longitude of the country.
</ParamField>

<ParamField query="unit" type="string" default="K">
  The unit of measurement to use. Possible values are:
  <ul>
    <li>K - Kilometers</li>
    <li>N - Nautical Miles</li>
  </ul>
</ParamField>
