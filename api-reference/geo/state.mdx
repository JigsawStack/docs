### Query

<ParamField query="country_code" type="string">
  The country code of the country. Filters result by country code.
</ParamField>

<ParamField query="state_code" type="string">
  The state code. Filters result by the state code.
</ParamField>

<ParamField query="search_value" type="string">
  The search value of the state.
</ParamField>

<ParamField query="lat" type="string">
  The latitude of the country. Required if `lng` is specified.
</ParamField>

<ParamField query="lng" type="string">
  The longitude of the country. Required if `lat` is specified.
</ParamField>

<ParamField query="limit" type="string" default="10">
  The maximum number of items to be returned.
</ParamField>
