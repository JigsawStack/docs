### Query

<ParamField query="country_code" type="string">
  The country code of the country. This filters the states to the country.
</ParamField>

<ParamField query="state_code" type="string">
  The state code of the state.
</ParamField>

<ParamField query="search_value" type="string">
  The search value of the state.
</ParamField>

<ParamField query="lat" type="string">
  The latitude of the country.
</ParamField>

<ParamField query="lng" type="string">
  The longitude of the country.
</ParamField>

<ParamField query="limit" type="string" default="10">
  The limit on the items to return
</ParamField>
