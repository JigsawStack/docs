---
title: "Password"
api: "GET https://api.jigsawstack.com/v1/generate/password"
description: "This endpoint is used generates a random password."
---

### Query

<ParamField query="type" type="string" default="easy_read">
  The type of password to generate. The default is easy_read.
  <ul>
    <li>easy_read</li>
    <li>easy_say</li>
    <li>all_characters</li>
  </ul>
</ParamField>

<ParamField query="length" type="string" default="16">
  The length of the password. The default is 16.
</ParamField>

<ParamField query="symbols" type="string" default="true">
  Whether to include symbols in the password. The default is true.
</ParamField>

<ParamField query="uppercase" type="string" default="true">
  Whether to include uppercase letters in the password. The default is true.
</ParamField>

<ParamField query="lowercase" type="string" default="true">
  Whether to include lowercase letters in the password. The default is true.
</ParamField>

<Snippet file="header.mdx" />

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="password" type="string">
  The generated password.
</ResponseField>
