### Query

<ParamField query="type" type="string" default="easy_read">
  The type of password to generate. Supported values:
  <ul>
    <li>`easy_read` - Avoids numbers and special characters.</li>
    <li>`easy_say` - Avoids ambiguous characters like l, 1, O, and 0.</li>
    <li>
      `all_characters` - Any character combinations like !, 7, h, K, and l1
    </li>
  </ul>
</ParamField>

<ParamField query="length" type="string" default="16">
  The length of the password.
</ParamField>

<ParamField query="symbols" type="string" default="true">
  Whether to include symbols in the password.
</ParamField>

<ParamField query="uppercase" type="string" default="true">
  Whether to include uppercase letters in the password.
</ParamField>

<ParamField query="lowercase" type="string" default="true">
  Whether to include lowercase letters in the password.
</ParamField>
