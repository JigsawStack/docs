---
title: "HTML To Image"
api: "POST https://api.jigsawstack.com/v1/generate/html2image"
description: "This endpoint is used to convert HTML to image"
---

### Body

<ParamField body="html" type="string" required>
  The HTML to convert to image.
</ParamField>

<ParamField body="full_page" type="boolean">
  Whether to take a full page screenshot.
</ParamField>

<ParamField body="omit_background" type="boolean">
  Whether to omit the background.
</ParamField>

<ParamField body="quality" type="number" default="75">
  The quality of the image. 0-100.
</ParamField>

<ParamField body="type" type="string" default="png">
  The type of the image. Can be one of the following:
  <ul>
    <li>jpeg</li>
    <li>png</li>
    <li>webp</li>
  </ul>
</ParamField>

<ParamField body="width" type="number" default="1920">
  The width of the screenshot.
</ParamField>

<ParamField body="height" type="number" default="1080">
  The height of the screenshot.
</ParamField>

<ParamField body="screen_size_preset" type="string">
  The screen size preset to use.
</ParamField>

<Snippet file="header.mdx" />
