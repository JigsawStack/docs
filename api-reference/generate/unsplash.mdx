### Query

<ParamField query="quality" type="string" default="reqular">
  The quality of the image. The default is regular.
  <ul>
    <li>raw</li>
    <li>full</li>
    <li>regular</li>
    <li>small</li>
    <li>thumb</li>
  </ul>
</ParamField>

<ParamField query="search_value" type="string">
  The search value of the image.
</ParamField>

<ParamField query="orientation" type="string">
  The orientation of the image.
  <ul>
    <li>landscape</li>
    <li>portrait</li>
    <li>squarish</li>
  </ul>
</ParamField>

<ParamField query="content_filter" type="string">
  The content filter of the image.
  <ul>
    <li>low</li>
    <li>high</li>
  </ul>
</ParamField>

<ParamField query="index" type="string">
  The index of the image.
  <ul>
    <li>0</li>
    <li>1</li>
    <li>2</li>
  </ul>
</ParamField>
