---
title: "Shorten URL"
api: "POST https://api.jigsawstack.com/v1/link"
description: "This endpoint is used to shorten a URL"
---

### Body

<ParamField body="url" type="string" required>
  The URL to shorten.
</ParamField>

<ParamField body="description" type="string">
  A description of the URL.
</ParamField>

<ParamField body="title" type="string">
  A title for the URL.
</ParamField>

<ParamField body="password" type="string">
  A password to protect the URL. If provided, the URL will only be accessible
  using the password
</ParamField>

<ParamField body="expires_at" type="string">
  A date string in the format YYYY-MM-DD.
</ParamField>

<ParamField body="slug" type="string">
  A custom slug for the URL.
</ParamField>

<ParamField body="geo_targeting" type="object">
Enables the targeting of URLs based on specific countries. For instance:

`{
"US": "https://www.google.us",
"CA": "https://www.google.ca",
}`

<Tip>Use valid `ISO2 country codes` as object key.</Tip>
</ParamField>
<ParamField body="redirect_domain" type="string">
Define custom domain for redirect. Defaults to `jigs.sh`
</ParamField>
### Header

<ParamField header="x-api-key" type="string" required>
  Your JigsawStack API key
</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="data" type="object">

<Expandable title="Toggle data">

<ResponseField name="url" type="string">
  The URL that was shortened.
</ResponseField>

<ResponseField name="slug" type="string">
  The slug of the URL.
</ResponseField>

<ResponseField name="short_url" type="string">
  The shortened URL.
</ResponseField>

</Expandable>

</ResponseField>

