title: "Start Chat Session"
api: "POST https://api.jigsawstack.com/v1/ai/chat"
description: "This endpoint is used to start a chat session"
---

### Body

<ParamField body="system_message" type="string">
  System message to start the chat session
</ParamField>

<ParamField body="document_ids" type="array<string">
  Document ids to give context to the chat session
</ParamField>

<Snippet file="header.mdx" />

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="chat_session_id" type="string">
  Chat session ID for subsequent chat interactions.
</ResponseField>
