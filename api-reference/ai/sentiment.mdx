---
title: "Sentiment"
api: "POST https://api.jigsawstack.com/v1/ai/sentiment"
description: "Perform sentiment analysis on a given text leveraging artificial intelligence."
---

### Body

<ParamField body="text" type="string" required>
  The text to be analyzed
</ParamField>

<Snippet file="header.mdx" />

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="sentiment" type="object">
<Expandable title="Toggle object" defaultOpen="true">
<ResponseField name="sentiment" type="string">
  Indicates the sentiment the text. E.g `positive` or `negative`
</ResponseField>
<ResponseField name="emotion" type="string">
  Indicates the type of emotion inferred from the text. E.g `happiness`
</ResponseField>
<ResponseField name="score" type="number">
  Indicates the sentiment score of the text.
</ResponseField>

</Expandable>

</ResponseField>

const endpoint = "https://api.jigsawstack.com/v1/ai/sentiment";
