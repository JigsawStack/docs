---

### Query

<ParamField query="email" type="string" required>
  The email address to validate.
</ParamField>

<Snippet file="header.mdx" />

### Response

<ResponseField name="valid" type="boolean">
  Indicates whether the email address has a valid syntax.
</ResponseField>

<ResponseField name="disposable" type="boolean">
  Indicates whether the email address is disposable.
</ResponseField>

<ResponseField name="role_account" type="boolean">
  Indicates whether the email address is a role_account.
</ResponseField>

<ResponseField name="free" type="boolean">
  Indicates whether the email address is a free email.
</ResponseField>

<ResponseField name="has_mx_records" type="boolean">
  Indicates whether the email address has MX records.
</ResponseField>
