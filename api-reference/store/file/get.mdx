<RequestExample>

```javascript Node.js
import { JigsawStack } from "jigsawstack";

const jigsawstack = JigsawStack({ apiKey: "your-api-key" });

const key = "file.png";

const result = await jigsawstack.store.retrieve(key);
const data = result.blob();
```
