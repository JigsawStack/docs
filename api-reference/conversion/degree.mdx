### Supported Degree

<Info>celcius | farenheit | kelvin</Info>

### Query

<ParamField query="from" type="string" required>
  The degree to convert from.
<ul>
  <li>celcius</li>
  <li>farenheit</li>
  <li>kelvin</li>
</ul>

</ParamField>

<ParamField query="to" type="string" required>
  The degree to convert to.

<ul>
  <li>celcius</li>
  <li>farenheit</li>
  <li>kelvin</li>
</ul>

</ParamField>

<ParamField query="value" type="string" required>
  The value to convert.
</ParamField>

<ResponseField name="success" type="boolean">

<ResponseField name="degree" type="number">
