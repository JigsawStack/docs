---
title: "QR Code"
api: "GET https://api.jigsawstack.com/api/v1/conversion/qrcode"
description: "This endpoint is used to generate QR code."
---

### Query

<ParamField query="text" type="string" required>
  The text to be encoded in the QR code.
</ParamField>

<ParamField query="height" type="string" default="512">
  The height of the QR code.
</ParamField>

<ParamField query="width" type="string" default="512">
  The width of the QR code.
</ParamField>

<ParamField query="background_color" type="string" default="#fff">
  The background color of the QR code.
</ParamField>

<ParamField query="qr_color" type="string" default="#000000ff">
  The color of the QR code.
</ParamField>

<ParamField query="type" type="string" default="jpeg">
  The type of the QR code. Can be either jpeg or png.
</ParamField>

<ResponseField name="success" type="boolean">
