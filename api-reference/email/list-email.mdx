### Query

<ParamField query="limit" default="30" type="string">
  The number of emails to return.
</ParamField>

<ParamField query="page" default="1" type="string">
  The page to return.
</ParamField>

<ParamField query="bulk_id" type="string">
  The bulk id to filter emails.
</ParamField>

<ParamField body="tags" type="array<object> | array<string>">
  Email tags. Maximum is `10`. Example `[{"name":"marketing"}]` or `["name"]`
</ParamField>

<ParamField query="broadcast_id" type="string">
  The broadcast id to filter emails.
</ParamField>
